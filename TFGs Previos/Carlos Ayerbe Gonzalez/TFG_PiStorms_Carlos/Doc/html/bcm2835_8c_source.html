212 /* Function select
213 // pin is a BCM2835 GPIO pin number NOT RPi pin number
214 // There are 6 control registers, each control the functions of a block
215 // of 10 pins.
216 // Each control register has 10 sets of 3 bits per GPIO pin:
217 //
218 // 000 = GPIO Pin X is an input
219 // 001 = GPIO Pin X is an output
220 // 100 = GPIO Pin X takes alternate function 0
221 // 101 = GPIO Pin X takes alternate function 1
222 // 110 = GPIO Pin X takes alternate function 2
223 // 111 = GPIO Pin X takes alternate function 3
224 // 011 = GPIO Pin X takes alternate function 4
225 // 010 = GPIO Pin X takes alternate function 5
226 //
227 // So the 3 bits for port X are:
228 // X / 10 + ((X % 10) * 3)
229 */
230 void bcm2835_gpio_fsel(uint8_t pin, uint8_t mode)
231 {
232  /* Function selects are 10 pins per 32 bit word, 3 bits per pin */
233  volatile uint32_t* paddr = bcm2835_gpio + BCM2835_GPFSEL0/4 + (pin/10);
234  uint8_t shift = (pin % 10) * 3;
235  uint32_t mask = BCM2835_GPIO_FSEL_MASK << shift;
236  uint32_t value = mode << shift;
237  bcm2835_peri_set_bits(paddr, value, mask);
238 }

515 /* Set the pullup/down resistor for a pin
516 //
517 // The GPIO Pull-up/down Clock Registers control the actuation of internal pull-downs on
518 // the respective GPIO pins. These registers must be used in conjunction with the GPPUD
519 // register to effect GPIO Pull-up/down changes. The following sequence of events is
520 // required:
521 // 1. Write to GPPUD to set the required control signal (i.e. Pull-up or Pull-Down or neither
522 // to remove the current Pull-up/down)
523 // 2. Wait 150 cycles ? this provides the required set-up time for the control signal
524 // 3. Write to GPPUDCLK0/1 to clock the control signal into the GPIO pads you wish to
525 // modify ? NOTE only the pads which receive a clock will be modified, all others will
526 // retain their previous state.
527 // 4. Wait 150 cycles ? this provides the required hold time for the control signal
528 // 5. Write to GPPUD to remove the control signal
529 // 6. Write to GPPUDCLK0/1 to remove the clock
530 //
531 // RPi has P1-03 and P1-05 with 1k8 pullup resistor
532 */
533 void bcm2835_gpio_set_pud(uint8_t pin, uint8_t pud)
534 {
535  bcm2835_gpio_pud(pud);
536  delayMicroseconds(10);
537  bcm2835_gpio_pudclk(pin, 1);
538  delayMicroseconds(10);
539  bcm2835_gpio_pud(BCM2835_GPIO_PUD_OFF);

637 void bcm2835_spi_transfernb(char* tbuf, char* rbuf, uint32_t len)
638 {
639  volatile uint32_t* paddr = bcm2835_spi0 + BCM2835_SPI0_CS/4;
640  volatile uint32_t* fifo = bcm2835_spi0 + BCM2835_SPI0_FIFO/4;
641  uint32_t TXCnt=0;

828 /* Writes an number of bytes to I2C */
829 uint8_t bcm2835_i2c_write(const char * buf, uint32_t len)
830 {
831 #ifdef I2C_V1
832  volatile uint32_t* dlen = bcm2835_bsc0 + BCM2835_BSC_DLEN/4;
833  volatile uint32_t* fifo = bcm2835_bsc0 + BCM2835_BSC_FIFO/4;
834  volatile uint32_t* status = bcm2835_bsc0 + BCM2835_BSC_S/4;
835  volatile uint32_t* control = bcm2835_bsc0 + BCM2835_BSC_C/4;
836 #else
837  volatile uint32_t* dlen = bcm2835_bsc1 + BCM2835_BSC_DLEN/4;
838  volatile uint32_t* fifo = bcm2835_bsc1 + BCM2835_BSC_FIFO/4;
839  volatile uint32_t* status = bcm2835_bsc1 + BCM2835_BSC_S/4;
840  volatile uint32_t* control = bcm2835_bsc1 + BCM2835_BSC_C/4;
841 #endif
842 
843  uint32_t remaining = len;
844  uint32_t i = 0;
845  uint8_t reason = BCM2835_I2C_REASON_OK;
846 
847  /* Clear FIFO */
848  bcm2835_peri_set_bits(control, BCM2835_BSC_C_CLEAR_1 , BCM2835_BSC_C_CLEAR_1 );
849  /* Clear Status */
850  bcm2835_peri_write(status, BCM2835_BSC_S_CLKT | BCM2835_BSC_S_ERR | BCM2835_BSC_S_DONE);
851  /* Set Data Length */
852  bcm2835_peri_write(dlen, len);
853  /* pre populate FIFO with max buffer */
854  while( remaining && ( i < BCM2835_BSC_FIFO_SIZE ) )
855  {
856  bcm2835_peri_write_nb(fifo, buf[i]);
857  i++;
858  remaining--;
859  }
860 
861  /* Enable device and start transfer */
862  bcm2835_peri_write(control, BCM2835_BSC_C_I2CEN | BCM2835_BSC_C_ST);
863 
864  /* Transfer is over when BCM2835_BSC_S_DONE */
865  while(!(bcm2835_peri_read(status) & BCM2835_BSC_S_DONE ))
866  {
867  while ( remaining && (bcm2835_peri_read(status) & BCM2835_BSC_S_TXD ))
868  {
869  /* Write to FIFO */
870  bcm2835_peri_write(fifo, buf[i]);
871  i++;
872  remaining--;
873  }
874  }
875 
876  /* Received a NACK */
877  if (bcm2835_peri_read(status) & BCM2835_BSC_S_ERR)
878  {
879  reason = BCM2835_I2C_REASON_ERROR_NACK;
880  }
881 
882  /* Received Clock Stretch Timeout */
883  else if (bcm2835_peri_read(status) & BCM2835_BSC_S_CLKT)
884  {
885  reason = BCM2835_I2C_REASON_ERROR_CLKT;
886  }
887 
888  /* Not all data is sent */
889  else if (remaining)
890  {
891  reason = BCM2835_I2C_REASON_ERROR_DATA;
892  }
893 
894  bcm2835_peri_set_bits(control, BCM2835_BSC_S_DONE , BCM2835_BSC_S_DONE);
895 
896  return reason;
897 }

1088  bcm2835_peri_write(status, BCM2835_BSC_S_CLKT | BCM2835_BSC_S_ERR | BCM2835_BSC_S_DONE);
1089

1091  bcm2835_peri_write(dlen, cmds_len);

1094  while( remaining && ( i < BCM2835_BSC_FIFO_SIZE ) )

1097  i++;
1098  remaining--;
1099  }

1102  bcm2835_peri_write(control, BCM2835_BSC_C_I2CEN | BCM2835_BSC_C_ST);
1103

1110  }
1111

1113  i = 0;
1114

1116  bcm2835_peri_write(dlen, buf_len);
1117  bcm2835_peri_write(control, BCM2835_BSC_C_I2CEN | BCM2835_BSC_C_ST | BCM2835_BSC_C_READ );
1118

1121

1124  {

1127  {

1129  buf[i] = bcm2835_peri_read_nb(fifo);
1130  i++;
1131  remaining--;
1132  }
1133  }
1134

1136  while (remaining && (bcm2835_peri_read(status) & BCM2835_BSC_S_RXD))
1137  {
1138  /* Read from FIFO */
1139  buf[i] = bcm2835_peri_read(fifo);
1140  i++;
1141  remaining--;
1142  }
1143

1145  if (bcm2835_peri_read(status) & BCM2835_BSC_S_ERR)
1146  {
1147  reason = BCM2835_I2C_REASON_ERROR_NACK;
1148  }
1149 
1150  /* Received Clock Stretch Timeout */
1151  else if (bcm2835_peri_read(status) & BCM2835_BSC_S_CLKT)
1152  {
1153  reason = BCM2835_I2C_REASON_ERROR_CLKT;
1154  }
1155 
1156  /* Not all data is sent */
1157  else if (remaining)
1158  {
1159  reason = BCM2835_I2C_REASON_ERROR_DATA;
1160  }
1161 
1162  bcm2835_peri_set_bits(control, BCM2835_BSC_S_DONE , BCM2835_BSC_S_DONE);
1163 
1164  return reason;
1165 }
